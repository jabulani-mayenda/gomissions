SQL Setup Checklist

Ensure you have created the storage buckets and tables in your Supabase project.

1

Create Storage Bucket

In Supabase, go to Storage -> Create a new bucket named gallery. Make sure it is marked as Public.

2

Run SQL Script

Go to SQL Editor and run:

-- 1. Create Posts Table
create table posts (
  id uuid primary key default gen_random_uuid(),
  type text not null,
  section text default 'gallery',
  image_url text,
  caption text,
  verse_text text,
  verse_ref text,
  posted_by text default 'GMI Team',
  created_at timestamptz default now()
);

-- 2. Create Volunteers Table
create table volunteers (
  id uuid primary key default gen_random_uuid(),
  name text,
  phone text,
  role text,
  created_at timestamptz default now()
);

-- 3. Enable RLS (Row Level Security)
alter table posts enable row level security;
alter table volunteers enable row level security;

-- 4. Setup Basic Policies
-- Allow public to view posts
create policy "Public read posts" on posts for select using (true);
-- Allow public to insert volunteers (from website form)
create policy "Public insert volunteers" on volunteers for insert with check (true);
-- Allow admin to manage posts (replace with auth logic later)
create policy "Admin manage posts" on posts for all using (true) with check (true);
